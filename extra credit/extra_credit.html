  1. (3 points) Are there any hotspots for shark incidents? Make a plot that shows the total number of incidents for the top 10 countries? Which country has the highest number of incidents?
global_sharks %>%
  group_by(country) %>%
  summarise(incident_total = n()) %>%
  top_n(10) %>%
  arrange(incident_total)%>%
  ggplot(aes(x=country,y=incident_total))+
  geom_col()+
  labs(title = "Top 10 Countries with Shark Incidents",
    x="Country",
    y="Incident Number")+
  coord_flip()
